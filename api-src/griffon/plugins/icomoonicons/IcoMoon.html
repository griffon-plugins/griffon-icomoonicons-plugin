499     @Nonnull
500     public static String asResource(@Nonnull String description) {
501         int size = 16;
502         checkDescription(description);
503 
504         String[] parts = description.split(":");
505         if (parts.length == 2) {
506             try {
507                 size = Integer.parseInt(parts[1]);
508             } catch (NumberFormatException e) {
509                 throw invalidDescription(description, e);
510             }
511         }
512 
513         IcoMoon feed = findByDescription(description, size);
514         return feed.asResource(size);
515     }
516 
517     @Nonnull
518     public static IcoMoon findByDescription(@Nonnull String description) {
519         checkDescription(description);
520 
521         IcoMoon feed = null;
522         String[] parts = description.split(":");
523         for (IcoMoon f : values()) {
524             if (f.description.equalsIgnoreCase(parts[0])) {
525                 feed = f;
526                 break;
527             }
528         }
529 
530         if (feed == null) {
531             throw invalidDescription(description);
532         }
533 
534         ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
535         if (parts.length == 2) {
536             int size = 16;
537             try {
538                 size = Integer.parseInt(parts[1]);
539             } catch (NumberFormatException e) {
540                 throw invalidDescription(description, e);
541             }
542             if (classLoader.getResource(feed.asResource(size)) != null) {
543                 return feed;
544             }
545         }
546 
547         if (classLoader.getResource(feed.asResource(16)) != null) {
548             return feed;
549         } else if (classLoader.getResource(feed.asResource(32)) != null) {
550             return feed;
551         }
552 
553         throw invalidDescription(description);
554     }
555 
556     @Nonnull
557     public static IcoMoon findByDescription(@Nonnull String description, int size) {
558         checkDescription(description);
559 
560         IcoMoon feed = null;
561         String[] parts = description.split(":");
562         for (IcoMoon f : values()) {
563             if (f.description.equalsIgnoreCase(parts[0])) {
564                 feed = f;
565                 break;
566             }
567         }
568 
569         if (feed == null) {
570             throw invalidDescription(description);
571         }
572 
573         ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
574         if (classLoader.getResource(feed.asResource(size)) != null) {
575             return feed;
576         }
577 
578         throw invalidDescription(description);
579     }

581     public static int requireValidSize(int size) {
582         requireState(size == 16 || size == 32, "Argument 'size' must be either 16 or 32.");
583         return size;
584     }
585 
586     private static void checkDescription(String description) {
587         if (isBlank(description)) {
588             throw invalidDescription(description);
589         }
590     }
591 
592     @Nonnull
593     public static IllegalArgumentException invalidDescription(@Nonnull String description) {
594         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
595         throw new IllegalArgumentException("Description " + description + " is not a valid IcoMoon icon description");
596     }
597 
598     @Nonnull
599     public static IllegalArgumentException invalidDescription(@Nonnull String description, Exception e) {
600         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
601         throw new IllegalArgumentException("Description " + description + " is not a valid IcoMoon icon description", e);
602     }
